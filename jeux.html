function selectAnswer(selectedIndex) { const question = questions[currentQuestionIndex]; const options = optionsElement.children; // Désactiver toutes les options for (let i = 0; i < options.length; i++) { options[i].style.pointerEvents = 'none'; } // Vérifier la réponse if (selectedIndex === question.correctIndex) { // Bonne réponse options[selectedIndex].style.background = 'rgba(0, 255, 0, 0.2)'; score += 1; } else { // Mauvaise réponse options[selectedIndex].style.background = 'rgba(255, 0, 0, 0.2)'; options[question.correctIndex].style.background = 'rgba(0, 255, 0, 0.2)'; } // Passer à la question suivante après un délai setTimeout(() => { currentQuestionIndex++; if (currentQuestionIndex < questions.length) { showQuestion(); } else { endGame(); } }, 1500); } function endGame() { // Arrêter le timer clearInterval(timerInterval); // Calculer le score final const finalScore = Math.round((score / questions.length) * 100); finalScoreElement.textContent = finalScore; finalProgressElement.style.width = `${finalScore}%`; // Personnaliser le message de résultat avec le nom if (finalScore >= 70) { successMessage.style.display = 'block'; failureMessage.style.display = 'none'; resultTitleElement.textContent = `Félicitations ${playerName}!`; resultDescriptionElement.textContent = `Incroyable ${playerName}! Tu as vraiment fait attention à moi...`; } else { successMessage.style.display = 'none'; failureMessage.style.display = 'block'; resultTitleElement.textContent = `Dommage ${playerName}...`; resultDescriptionElement.textContent = `${playerName}, tu as obtenu ${finalScore}%. Il te manque ${70 - finalScore}% pour réussir mon défi. Je suis sûre que tu peux faire mieux!`; } // Afficher l'écran de résultat quizScreen.style.display = 'none'; resultScreen.style.display = 'block'; } function restartGame() { // Réinitialiser le formulaire de nom playerNameInput.value = ""; resultScreen.style.display = 'none'; startScreen.style.display = 'flex'; } function shareScore() { const finalScore = Math.round((score / questions.length) * 100); alert(`${playerName} a obtenu ${finalScore}% au Défi de Séduction de Pine Nikeuse!
